-- Create a new database for chapters
CREATE DATABASE IF NOT EXISTS chapter_db;
USE chapter_db;

-- Create chapters table
DROP TABLE IF EXISTS chapter_pages;
DROP TABLE IF EXISTS chapter;

CREATE TABLE IF NOT EXISTS chapter (
    chapter_id INT AUTO_INCREMENT PRIMARY KEY,
    comic_id INT NOT NULL,
    chapter_number INT NOT NULL,
    title VARCHAR(255) NOT NULL,
    release_date DATE NOT NULL,
    UNIQUE KEY unique_chapter (comic_id, chapter_number)
);

-- Create chapter pages table to store multiple images per chapter
CREATE TABLE IF NOT EXISTS chapter_pages (
    page_id INT AUTO_INCREMENT PRIMARY KEY,
    comic_id INT NOT NULL,
    chapter_id INT NOT NULL,
    page_number INT NOT NULL,
    page_image LONGBLOB NOT NULL,
    UNIQUE KEY unique_page (comic_id, chapter_id, page_number),
    FOREIGN KEY (chapter_id) REFERENCES chapter(chapter_id) ON DELETE CASCADE
);

-- Verify chapters and their pages were loaded correctly
SELECT 
    c.chapter_id,
    c.comic_id,
    co.comic_name,
    c.chapter_number,
    c.title,
    c.release_date,
    COUNT(cp.page_id) as page_count
FROM chapter c
LEFT JOIN comic_db.comic co ON c.comic_id = co.comic_id
LEFT JOIN chapter_pages cp ON c.chapter_id = cp.chapter_id
GROUP BY c.chapter_id
ORDER BY c.comic_id, c.chapter_number;
